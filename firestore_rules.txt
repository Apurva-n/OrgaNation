rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Authenticated users can read/write donor data
    match /donors/{donorId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write recipient data
    match /Recepients/{recipientId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write hospital data
    match /hospitals/{hospitalId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write admin data
    match /Admin/{adminId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write organ requests
    match /organ_requests/{requestId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write recipient requests
    match /recipient_requests/{requestId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write notifications
    match /notifications/{notificationId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write email notifications
    match /email_notifications/{notificationId} {
      allow read, write: if request.auth != null;
    }
    
    // Authenticated users can read/write transplant notifications
    match /transplant_notifications/{notificationId} {
      allow read, write: if request.auth != null;
    }
  }
}
